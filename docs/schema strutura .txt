create table public.appointments (
  id uuid not null default gen_random_uuid (),
  client_id uuid not null,
  staff_id uuid not null,
  service_id uuid not null,
  starts_at timestamp with time zone not null,
  ends_at timestamp with time zone not null,
  status public.appointment_status not null default 'pending'::appointment_status,
  price numeric(10, 2) not null,
  notes text null,
  cancelled_at timestamp with time zone null,
  cancelled_reason text null,
  created_by uuid null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  payment_method_id uuid null,
  constraint appointments_pkey primary key (id),
  constraint appointments_created_by_fkey foreign KEY (created_by) references profiles (id),
  constraint appointments_client_id_fkey foreign KEY (client_id) references clients (id) on delete CASCADE,
  constraint appointments_staff_id_fkey foreign KEY (staff_id) references staff (id) on delete RESTRICT,
  constraint appointments_service_id_fkey foreign KEY (service_id) references services (id) on delete RESTRICT,
  constraint chk_ends_after_starts check ((ends_at > starts_at)),
  constraint no_overlap_per_staff EXCLUDE using gist (
    staff_id
    with
      =,
      tstzrange (starts_at, ends_at, '[)'::text)
    with
      &&
  )
  where
    (
      (
        status <> all (
          array[
            'cancelled'::appointment_status,
            'no_show'::appointment_status
          ]
        )
      )
    )
) TABLESPACE pg_default;

create index IF not exists idx_appointments_client on public.appointments using btree (client_id) TABLESPACE pg_default;

create index IF not exists idx_appointments_staff on public.appointments using btree (staff_id) TABLESPACE pg_default;

create index IF not exists idx_appointments_staff_starts on public.appointments using btree (staff_id, starts_at) TABLESPACE pg_default;

create index IF not exists idx_appointments_starts on public.appointments using btree (starts_at) TABLESPACE pg_default;

create index IF not exists idx_appointments_status on public.appointments using btree (status) TABLESPACE pg_default;

create trigger trg_appointment_conflict BEFORE INSERT
or
update OF staff_id,
starts_at,
ends_at,
status on appointments for EACH row when (
  new.status = any (
    array[
      'pending'::appointment_status,
      'confirmed'::appointment_status,
      'completed'::appointment_status
    ]
  )
)
execute FUNCTION check_appointment_conflict ();

create trigger trg_appointment_lifecycle
after INSERT
or
update on appointments for EACH row
execute FUNCTION handle_appointment_lifecycle ();

create trigger trg_appointments_updated_at BEFORE
update on appointments for EACH row
execute FUNCTION set_updated_at ();

create trigger trg_maintain_client_metrics
after INSERT
or DELETE
or
update on appointments for EACH row
execute FUNCTION maintain_client_metrics ();
